\begin{misc}{Path to Guido's Cave}
    \begin{itemize}
        \item East until you hit a set of rocks
        \item North until you see rocks on the far left side of the screen
        \item West until you hit rocks
        \item North until you see the cave
    \end{itemize}
\end{misc}
